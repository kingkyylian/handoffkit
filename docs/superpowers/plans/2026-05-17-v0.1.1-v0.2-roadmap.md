# HandoffKit v0.1.1 and v0.2.0 Roadmap Implementation Plan

> **For agentic workers:** Execute this plan task-by-task. In Codex, use `executing-plans` inline unless the user explicitly asks for subagents or parallel agent work. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Turn the published v0.1.0 CLI into a trustworthy v0.1.1 maintenance release, then build v0.2.0 around the product differentiator: structured AI coding session resume packets.

**Architecture:** v0.1.1 is operational polish only: automate the release smoke path, improve first-run CLI errors/help, and move publishing toward GitHub Actions provenance. v0.2.0 adds product depth in three bounded modules: deterministic resume parsing, target-specific Markdown profiles, and local secret scanner guidance/config discovery. Keep collection local-first and deterministic; do not add LLM APIs, telemetry, background services, or broad repo ingestion.

**Tech Stack:** Node.js >=22, TypeScript ESM, commander, zod, execa, fast-glob, vitest, tsup, eslint, pnpm, GitHub Actions, npm publishing.

```js
#!/usr/bin/env node
import { mkdir, mkdtemp, rm, writeFile } from "node:fs/promises";
import { tmpdir } from "node:os";
import { join } from "node:path";
import { execa } from "execa";

const root = process.cwd();
const workDir = await mkdtemp(join(tmpdir(), "handoffkit-release-smoke-"));
const packDir = join(workDir, "pack");
const installDir = join(workDir, "install");

try {
  await mkdir(packDir, { recursive: true });
  await mkdir(installDir, { recursive: true });
  await run("npm", ["--cache", join(workDir, "npm-cache"), "pack", "--pack-destination", packDir], { cwd: root });
  const tarball = await newestTarball(packDir);

  await run("git", ["init", "--initial-branch=main"], { cwd: installDir });
  await writeFile(join(installDir, "README.md"), "# Smoke\n", "utf8");
  await run("npm", ["init", "-y"], { cwd: installDir });
  await run("npm", ["--cache", join(workDir, "npm-cache"), "install", tarball], { cwd: installDir });

  const pack = await jsonCommand(installDir, ["pack", "--goal", "Release smoke", "--format", "json", "--no-diff"]);
  assertNoGeneratedChangedFiles(pack.repository.changedFiles);

  await jsonCommand(installDir, ["risk", "--format", "json"]);
  await jsonCommand(installDir, ["scan-secrets", "--format", "json"]);
  await jsonCommand(installDir, ["resume", "README.md", "--goal", "Resume smoke", "--format", "json"]);

  console.log(`release smoke passed in ${installDir}`);
} finally {
  if (process.env.HANDOFFKIT_KEEP_SMOKE !== "1") {
    await rm(workDir, { recursive: true, force: true });
  }
}

async function newestTarball(directory) {
  const { readdir } = await import("node:fs/promises");
  const files = (await readdir(directory)).filter((file) => file.endsWith(".tgz")).sort();

  if (files.length !== 1) {
    throw new Error(`Expected one tarball in ${directory}, found ${files.length}.`);
  }

  return join(directory, files[0]);
}

async function jsonCommand(cwd, args) {
  const result = await run("./node_modules/.bin/handoffkit", args, { cwd });
  return JSON.parse(result.stdout);
}

async function run(command, args, options) {
  const result = await execa(command, args, {
    ...options,
    reject: false,
    all: true
  });

  if (result.exitCode !== 0) {
    throw new Error(`${command} ${args.join(" ")} failed:\n${result.all ?? result.stderr}`);
  }

  return result;
}

function assertNoGeneratedChangedFiles(files) {
  const generated = files.filter((file) =>
    file === "node_modules" ||
    file === "dist" ||
    file === "coverage" ||
    file.startsWith("node_modules/") ||
    file.startsWith("dist/") ||
    file.startsWith("coverage/")
  );

  if (generated.length > 0) {
    throw new Error(`Generated files leaked into changedFiles: ${generated.join(", ")}`);
  }
}
```

```ts
export function parseResumeState(content: string): ResumeState {
  const state: ResumeState = {
    completed: [],
    remaining: [],
    failedCommands: [],
    openQuestions: [],
    verification: []
  };
  let section: keyof ResumeState | undefined;
  let heading: string | undefined;

  for (const rawLine of content.replace(/\r\n/g, "\n").split("\n")) {
    const line = rawLine.trim();
    const headingMatch = line.match(/^#{1,4}\s+(.+)$/);

    if (headingMatch) {
      heading = headingMatch[1].trim();
      section = sectionForHeading(heading);
      continue;
    }

    if (!section || section === "nextSafestAction") {
      continue;
    }

    const item = normalizeListItem(line);
    if (item) {
      state[section].push({ text: redactText(item), ...(heading ? { sourceHeading: heading } : {}) });
    }
  }

  const next = state.remaining[0] ?? state.openQuestions[0] ?? state.failedCommands[0];
  if (next) {
    state.nextSafestAction = next.text;
  }

  return state;
}
```

Add helpers:

```ts
function sectionForHeading(heading: string): keyof ResumeState | undefined {
  for (const [section, patterns] of Object.entries(SECTION_ALIASES)) {
    if (patterns.some((pattern) => pattern.test(heading))) {
      return section as keyof ResumeState;
    }
  }

  return undefined;
}

function normalizeListItem(line: string) {
  const match = line.match(/^[-*]\s+(.+)$/) ?? line.match(/^\d+\.\s+(.+)$/);
  return match?.[1]?.trim();
}
```

function renderResumeState(state: HandoffReport["resumeSource"] extends infer Source ? Source extends { state?: infer State } ? State : never : never) {
  if (!state) {
    return [];
  }

  const resumeState = state as NonNullable<HandoffReport["resumeSource"]>["state"];
  if (!resumeState) {
    return [];
  }

  return [
    "## Resume State",
    renderResumeItems("Completed", resumeState.completed),
    renderResumeItems("Remaining", resumeState.remaining),
    renderResumeItems("Failed Commands", resumeState.failedCommands),
    renderResumeItems("Open Questions", resumeState.openQuestions),
    renderResumeItems("Verification", resumeState.verification),
    resumeState.nextSafestAction ? `- Next safest action: ${resumeState.nextSafestAction}` : "- Next safest action: none detected.",
    ""
  ];
}

function renderResumeItems(title: string, items: { text: string }[]) {

async function scannerConfigFiles(name: "gitleaks" | "secretlint", root: string) {
  const patterns =
    name === "gitleaks"
      ? ["gitleaks.toml", ".gitleaks.toml", ".config/gitleaks/*.toml"]
      : [".secretlintrc", ".secretlintrc.*", "secretlint.config.*"];

  return fg(patterns, {
    cwd: root,
    dot: true,
    onlyFiles: true,
    unique: true
  });
}

function installHint(name: "gitleaks" | "secretlint") {
  return name === "gitleaks"
    ? "Install gitleaks from https://github.com/gitleaks/gitleaks, then rerun with --scan-secrets."
    : "Install secretlint from https://github.com/secretlint/secretlint, then rerun with --scan-secrets.";
}
